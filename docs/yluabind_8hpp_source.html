133 if constexpr (idx < sizeof...(Targs)) {

152 TupleType t;
153 int top_of_stack = lua_gettop(l);
154 index = index < 0? top_of_stack + index + 1: index;
155 if (index <= top_of_stack) {
156 for (lua_Integer i = 1; i <= sizeof...(Targs); ++i) {
157 lua_geti(l, index, i);
